#include "unp.h"

int main(int argc, char* argv[])
{
    int sockfd, n;
    char recvline[MAXLENGTH + 1];
    struct sockaddr_in servaddr;
    if (argc != 2) err_quit("Usage:a.out <IPAddress>");
    if ((sockfd = socket(AF_INET, SOCK_STREAM, 0)) < 0) err_sys("socket error");
    bzero(&servaddr, sizeof(servaddr)); // 置0; memset(&servaddr, 0, sizeof(servaddr));
    servaddr.sin_family = AF_INET;
    servaddr.sin_port   = htons(13);  // 时间服务端口
    if (inet_pton(AF_INET, argv[1], &servaddr.sin_addr) <= 0) err_quit("inet_pton error for %s", argv[1]);
    if (connect(sockfd, (SA*)&servaddr, sizeof(servaddr)) < 0) err_sys("connect error");
    while ((n = read(socked, recvline, MAXLENGTH)) > 0)
    {
        recvline[n] = '\0';
        if (fputs(recvline, stdout) == EOF) err_sys("fputs error");
    }
    if (n < 0) err_sys("read error");
    return 0;
}

协议无关性

该时间访问程序是 IPv4 相关的, 要使用 IPv6 必须更改一些地方. 之后会学习协议无关的程序编写方法

包裹函数约定

一些操作需要检查错误码, 并在发生错误时打印错误信息并退出, 可以把这些语句放到一个函数中

Unix errno

一个 Unix 函数发生错误, 全局变量被设为指出该错误类型的正值, 函数本身返回-1. err_sys 查看 errno 并输出相应的错误信息. 当 errno 等于 ETIMEDOUT 时输出连接超时

#include "unp.h"
#include <time.h>
int main(int argc, char* argv[])
{
    int listenfd, connfd;
    SA servaddr;
    char buff[MAXLINE];
    time_t ticks;
    listenfd = Socket(AF_INET, SOCK_STREAM, 0);  // 打开套接字
    bzero(&servaddr, sizeof(servaddr));
    servaddr.sin_family      = AF_INET;
    servaddr.sin_addr.s_addr = htonl(INADDR_ANY);
    servaddr.sin_port        = htons(13);
    Bind(listenfd, (SA*)&servaddr, sizeof(servaddr));
    Listen(listenfd, LISTENQ);  // 变为监听套接字
    for (;;)
    {
        connfd = Accept(listenfd, (SA*)NULL, NULL);
        ticks  = time(NULL);
        snprintf(buff, MAXLINE, "%.24s\r\n", ctime(&ticks));
        Write(connfd, buff, strlen(buff));
        Close(connfd);
    }
    return 0;
}
