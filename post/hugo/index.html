// sudoku_solver.c
// 假设输入都是1到9的整数

#include <pthread.h>
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

typedef struct {
    int row;
    int col;
    int id;
} parameters;

int SUDOKU[9][9];
unsigned int flag = 0;
pthread_t TID[11];
void* rowChecker(void* params);
void* colChecker(void* params);
void* blockChecker(void* params);

int main(void)
{
    pthread_attr_t attr;
    printf("输入数独的解:\n");
    for (size_t i = 0; i != 9; ++i)
        for (size_t j = 0; j != 9; ++j)
        {
            scanf("%d", &SUDOKU[i][j]);
        }
    pthread_attr_init(&attr);
    for (size_t i = 0; i != 9; ++i)
    {
        parameters* data = (parameters*)malloc(sizeof(parameters));
        data->row        = i / 3 * 3;
        data->col        = i % 3 * 3;
        data->id         = i;
        pthread_create(&TID[i], &attr, blockChecker, data);
    }
    pthread_create(&TID[9], &attr, rowChecker, NULL);
    pthread_create(&TID[10], &attr, colChecker, NULL);
    for (int i = 0; i < 11; ++i)
    {
        pthread_join(TID[i], NULL);
    }
    if (flag)
    {
        printf("错误解%x\n", flag);
    }
    else
    {
        printf("验证通过\n");
    }
    return 0;
}

void* rowChecker(void* params)
{
    for (size_t i = 0; i != 9; ++i)
    {
        int a[10] = {0};
        for (size_t j = 0; j != 9; ++j)
        {
            if (!a[SUDOKU[i][j]]++)
                ;
            else
            {
                flag |= 1 << 9;
                pthread_exit(0);
            }
        }
    }
    pthread_exit(0);
}

void* colChecker(void* params)
{
    for (size_t i = 0; i != 9; ++i)
    {
        int a[10] = {0};
        for (size_t j = 0; j != 9; ++j)
        {
            if (!a[SUDOKU[j][i]]++)
                ;
            else
            {
                flag |= 1 << 10;
                pthread_exit(0);
            }
        }
    }
    pthread_exit(0);
}

void* blockChecker(void* params)
{
    int a[10] = {0};
    for (size_t i = 0; i != 3; ++i)
    {
        for (size_t j = 0; j != 3; ++j)
        {
            if (!a[SUDOKU[((parameters*)params)->row + i][((parameters*)params)->col + j]]++)
                continue;
            else
            {
                flag |= 1 << ((parameters*)params)->id;
                pthread_exit(0);
            }
        }
    }
    pthread_exit(0);
}
