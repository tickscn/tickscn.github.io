标准库

内置在 Go 中包含常用功能的包有 150 个以上, 学习一些简单的用途

  • unsafe: 一般用于 C/C++ 程序的调用中, 打破了 Go 的类型安全
  • syscall,os, os/exec:
    • syscall: 提供操作系统低层调用的基本接口
    • os: 提供平台无关的系统接口
    • os/exec: 运行外部指令
  • archive/xxx: 提供相应的压缩和解压命令
  • fmt, io, bufio, path/filepath
    • fmt: 格式化输入与输出
    • io: 基本的输入输出功能
    • bufio: 缓冲的输入输出
    • path/filepath: 路径
    • flag: 命令行参数的操作
    • strings, strconv, unicode, regexp, bytes
      • strings: 对字符串的操作
      • strconv: 字符串与其它类型的转换
      • unicode: unicode 字符支持
      • regexp: 正则表达式
      • bytes: 对字符分片的操作
      • index/suffixarray: 查询
    • math, math/cmath, math/rand, math/big, sort: 常用的数学函数, 对复数的操作, 随机数, 大数, 排序
    • container/xxx: 集合的操作, 包括双链表(list), 环形链表(ring), 堆(heap)
    • time
    • log
    • encoding/xxx: 支持 JSON, XML 等的转换
    • text/template: 模版生成
    • net, net/http, html
    • runtime: Go 运行时的相关操作
    • reflect: 反射

  "container/list"
  "fmt"
)

func main() {
  var l = new(list.List)
  l.PushBack(101)
  l.PushBack(102)
  l.PushBack(103)
  for e := l.Front(); e != nil; e = e.Next() {
    fmt.Println(e.Value)
  }
}

package main

import (
  "fmt"
  "os"
  "regexp"
)

const patternURL = "^((ht|f)tps?)://[\\w\\-]+(\\.[\\w\\-]+)+([\\w\\-\\.,@?^=%&:/~\\+#]*[\\w\\-\\@?^=%&/~\\+#])?$"

func main() {
  s := os.Args[1]
  if ok, _ := regexp.MatchString(patternURL, s); ok {
    fmt.Println(s, "is a valied URL")
  } else {
    fmt.Println(s, "is an invalied URL")
  }
}

type SyncBuf struct {
  Mutex sync.Mutex
  buf   bytes.Buffer
}

func UpdateSyncBuf(b *SyncBuf, s string) {
  b.Mutex.Lock()
  defer b.Mutex.Unlock()
  b.buf.WriteString(s)
}
