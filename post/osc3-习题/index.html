#include <stdio.h>
#include <stdlib.h>
#define MIN_PID 300
#define MAX_PID 5000
#define TRUE 1
#define FALSE 0

struct PidTable {
    int* pid;
    int size;
} PID;

int allocate_map(void)
{
    int i;
    PID.size = (MAX_PID - MIN_PID) / sizeof(int) + 1;
    PID.pid  = calloc(PID.size, sizeof(int));
    if (PID.pid == NULL) return -1;
    return 1;
}

int allocate_pid(void)
{
    int i;
    for (i = 0; i < PID.size; i++)
    {
        if (PID.pid[i] == -1) continue;
        for (int j = 0; j < sizeof(int); ++j)
        {
            if ((PID.pid[i] & (1 << j)) == 0)
            {
                PID.pid[i] += 1 << j;
                return i * sizeof(int) + j + MIN_PID;
            }
        }
    }
    return -1;
}

void release_pid(int pid)
{
    int i = (pid - MIN_PID) / sizeof(int);
    int j = (pid - MIN_PID) % sizeof(int);
    PID.pid[i] &= ~(1 << j);
}

int main()
{
    int pid;
    allocate_map();
    if ((pid = allocate_pid()) != -1) printf("New Process Allocated Pid= %d \n", pid);

    if ((pid = allocate_pid()) != -1) printf("New Process Allocated Pid= %d \n", pid);

    if ((pid = allocate_pid()) != -1) printf("New Process Allocated Pid= %d \n", pid);

    if ((pid = allocate_pid()) != -1) printf("New Process Allocated Pid= %d \n", pid);

    printf("Process %d now Releasing \n", pid);
    release_pid(pid);

    if ((pid = allocate_pid()) != -1) printf("New Process Allocated Pid= %d \n", pid);
    return 0;
}

#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>

int main(int argc, char* argv[])
{
    if (argc > 2) printf("参数只能为一个整数\n");
    pid_t pid;
    pid = fork();
    if (pid < 0)
    {
        fprintf(stderr, "Fork Failed");
        exit(1);
    }
    else if (pid == 0)
    {
        int i;
        sscanf(argv[1], "%d", &i);
        while (i != 1)
        {
            printf("%d ", i);
            if (i % 2 == 0)
                i >>= 1;
            else
                i = i * 3 + 1;
        }
        printf("1\n");
    }
    else
    {
        wait(NULL);
    }
    return 0;
}
