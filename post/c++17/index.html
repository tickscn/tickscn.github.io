#include <chrono>
#include <iostream>
using namespace std;
struct Foo {
    int val;
    Foo(int _val) : val(_val) {}
    auto getValV()
    {
        return [*this]() { return val; };
    }
    auto getValR()
    {
        return [this]() { return val; };
    }
};
int main()
{
  Foo a(42);
  auto c = a.getValV();
  auto d = a.getValR();
  a.val = 24;
  cout << c() <<endl;
  cout << d() << endl;
    return 0;
}

#include <bits/stdc++.h>
using namespace std;
pair<int, int> findMinandMax(vector<int>& vec)
{
    int m1 = INT_MAX, m2 = INT_MIN;
    for_each(vec.begin(), vec.end(), [&](auto v) {
        m1 = min(v, m1);
        m2 = max(v, m2);
    });
  return make_pair(m1, m2);
}
int main()
{
  default_random_engine e;
  uniform_int_distribution<int> u(0,100);
  vector<int> v;
  for(int i = 0; i < 100 ;++i) v.push_back(u(e));
  auto [minV, maxV] = findMinandMax(v);
  cout << minV << endl;
  cout << maxV << endl;
    return 0;
}
