package main

import (
  "fmt"
  "os"
  "strconv"
)

func main() {
  var grade byte = 'A'
  var marks int64 = 90
  marks, _ = strconv.ParseInt(os.Args[1], 10, 64)
  switch {
  case marks >= 90:
    grade = 'A'
  case marks >= 80:
    grade = 'B'
  case marks >= 60:
    grade = 'C'
  default:
    grade = 'D'
  }

  switch grade {
  case 'A':
    fmt.Printf("优秀\n")
  case 'B':
    fmt.Printf("良好\n")
  case 'C':
    fmt.Printf("一般\n")
  case 'D':
    fmt.Printf("差\n")
  default:
    fmt.Printf("不行\n")
  }
  fmt.Printf("你的等级是 %c\n", grade)
}

switch 可以用来判断类型

select 语句

随机执行一个可匹配的语句, 如果没有可执行的块, 它会阻塞到有 case 可运行. 每个 case 必须是一个通信. Go 不会重新对 channel 求值

package main

import (
  "fmt"
)

func main() {
  for i := 1; i < 10; i++ {
    for j := 1; j <= i; j++ {
      fmt.Printf("%dx%d=%-2d ", j, i, j*i)
    }
    fmt.Println()
  }
}

package main

import (
  "fmt"
  "math/rand"
  "time"
)

func main() {
  rand.Seed(time.Now().Unix())
  guess := rand.Intn(100) + 1
  var res int
  count := 0
  for {
    count++
    fmt.Scanln(&res)
    if res < guess {
      fmt.Println("小了")
    } else if res > guess {
      fmt.Println("大了")
    } else {
      break
    }
  }
  fmt.Printf("一共猜了 %d次, 答案是 %d\n", count, guess)
}

package main

import (
  "fmt"
)

func main() {
  /*
     1900-1-1 是星期一
  */
  var year, month int
  fmt.Printf("input year:\nyear = ")
  fmt.Scanf("%d", &year)
  fmt.Printf("input month:\nmonth = ")
  fmt.Scanf("%d", &month)

  sumDays := 1

  for i := 1900; i < year; i++ {
    if i%4 == 0 && i%100 != 0 || i%400 == 0 {
      sumDays += 366
    } else {
      sumDays += 365
    }
  }

  for i := 1; i < month; i++ {
    switch i {
    case 1, 3, 5, 7, 8, 10:
      sumDays += 31
    case 2:
      if year%4 == 0 && year%100 != 0 || year%400 == 0 {
        sumDays += 29
      } else {
        sumDays += 28
      }
    default:
      sumDays += 30
    }
  }
  var ttl int
  switch month {
  case 1, 3, 5, 7, 8, 10:
    ttl = 31
  case 2:
    if year%4 == 0 && year%100 != 0 || year%400 == 0 {
      ttl = 29
    } else {
      ttl = 28
    }
  default:
    ttl = 30
  }
  fmt.Println("total days ", sumDays)
  fmt.Println("Mon Tue Wed Thu Fri Sat Sun")
  for i := 0; i < (sumDays-1)%7; i++ {
    fmt.Printf("    ")
  }
  for i, j := 1, sumDays%7; i <= ttl; i++ {
    fmt.Printf("%3d ", i)
    if j%7 == 0 {
      fmt.Println()
    }
    j = (j + 1) % 7
  }
  fmt.Println()
}
