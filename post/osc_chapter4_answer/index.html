/************************ 3.13 ****************************/
#include <pthread.h>
#include <stdio.h>
#include <stdlib.h>
#include <time.h>
#include <unistd.h>
#define MIN_PID 300
#define MAX_PID 5000
#define THREADS_NUM 100

struct PidTable {
    int* pid;
    int size;
} PID;

int allocate_map(void)
{
    PID.size = (MAX_PID - MIN_PID) / sizeof(int) + 1;
    PID.pid  = calloc(PID.size, sizeof(int));
    if (PID.pid == NULL) return -1;
    return 1;
}

int allocate_pid(void)
{
    int i;
    for (i = 0; i < PID.size; i++)
    {
        if (PID.pid[i] == -1) continue;
        for (size_t j = 0; j < sizeof(int); ++j)
        {
            if ((PID.pid[i] & (1 << j)) == 0)
            {
                PID.pid[i] += 1 << j;
                return i * sizeof(int) + j + MIN_PID;
            }
        }
    }
    return -1;
}

void release_pid(int pid)
{
    int i = (pid - MIN_PID) / sizeof(int);
    int j = (pid - MIN_PID) % sizeof(int);
    PID.pid[i] &= ~(1 << j);
}

/************************ 4.15 ****************************/
void* runner(void* param);
int main(void)
{
    pthread_t tids[THREADS_NUM];
    pthread_attr_t attr;
    srand((unsigned int)time(NULL));
    allocate_map();
    pthread_attr_init(&attr);
    for (size_t i = 0; i != THREADS_NUM; ++i)
    {
        pthread_create(&tids[i], &attr, runner, NULL);
    }

    for (size_t i = 0; i < THREADS_NUM; ++i)
    {
        pthread_join(tids[i], NULL);
    }
    return 0;
}

void* runner(void* param)
{
    int pid = allocate_pid();
    int t   = rand() % 60;
    sleep(t);
    printf("The pid  %d is cost %d minutes", pid, t);
    release_pid(pid);
    pthread_exit(0);
}

#include <limits.h>
#include <pthread.h>
#include <stdio.h>
#include <stdlib.h>

double average = 0.0;
int mimimum    = INT_MAX;
int maximum    = INT_MIN;
int* values;

void* ave(void* param);
void* min(void* param);
void* max(void* param);
int main(int argc, char* argv[])
{
    pthread_t tids[3];
    pthread_attr_t attr;
    int num;

    if (argc < 2)
    {
        fprintf(stderr, "Usage: a.out <integer values>\n");
        return -1;
    }
    num    = argc - 1;
    values = malloc(num * sizeof(num));
    for (size_t i = 0; i != num; ++i)
    {
        values[i] = atoi(argv[i + 1]);
    }
    pthread_attr_init(&attr);
    pthread_create(&tids[0], &attr, ave, &num);
    pthread_create(&tids[1], &attr, min, &num);
    pthread_create(&tids[2], &attr, max, &num);

    for (size_t i = 0; i != 3; ++i) pthread_join(tids[i], NULL);

    printf("The average valuse is %f\n", average);
    printf("The mimimum valuse is %d\n", mimimum);
    printf("The maximum valuse is %d\n", maximum);

    free(values);
    return 0;
}

void* ave(void* param)
{
    int sum = 0;
    for (size_t i = 0; i != *(int*)param; ++i)
    {
        sum += values[i];
    }
    average = sum / (*((int*)param));
}

void* min(void* param)
{
    for (size_t i = 0; i != *(int*)param; ++i)
    {
        if (values[i] < mimimum) mimimum = values[i];
    }
}

void* max(void* param)
{
    for (size_t i = 0; i != *(int*)param; ++i)
    {
        if (values[i] > maximum) maximum = values[i];
    }
}

#include <omp.h>
#include <stdbool.h>
#include <stdio.h>
#include <stdlib.h>
#include <time.h>

const int POINTS_NUM = 10000;
const int LENGTH     = (1 << 10) + 1;
const int RADIO      = 1 << 9;
const int RADIO_2    = 1 << 18;

int main(void)
{
    int incircle = 0;
    int x, y;
    double pi;
    srand(time(NULL));
#pragma omp parallel for
    for (size_t i = 0; i != POINTS_NUM; ++i)
    {
        x = rand() % LENGTH - RADIO;
        y = rand() % LENGTH - RADIO;
        if (x * x + y * y <= RADIO_2) ++incircle;
    }

    pi = (double)incircle * 4 / POINTS_NUM;
    printf("pi = %f\n", pi);
    return 0;
}
