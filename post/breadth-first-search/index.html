广度优先搜索

给定图 G=(V,E) 的一个可以识别的源结点 s, 计算从s可到达的结点的距离, 同时生成一个广度优先搜索树

BFS(G, s)
    for each vertex u in G.V - {s}
        u.color = WHITE;
        u.d = inf;
        u.pre = NIL;
    s.color = GRAY;
    s.d = 0;
    s.pre = GRAY;
    Q = empty queue;
    ENQUEUE(Q, s);
    while(!Q.empty())
        u = DEQUEUE(Q);
        for each v in G.Adj[u]
            if v.color == WHITE
                v.color = GRAY;
                v.d = u.d + 1;
                v.pre = u;
                ENQUEUE(Q, v);
        u.color = BLACK;

DFS(G) //< 递归
    for each vertex u in G.V
        u.color = WHITE
        u.pre = NIL
    time = 0
    for each vertex u in G.V
        if u.color = WHITE
            VISIT(G, u)
            
VISIT(G, u)
    time = time + 1
    u.d = time //< 第一次访问时间
    u.color = GRAY;
    for each v in G.Adj[u]
        if v.color = WHITE
            v.pre = u
            VISIT(G,v)
    u.color = BLACK
    time = time + 1
    u.f = time //< 访问完的时间

GFS(G) // 迭代
    for each vertex u in G.V
        u.color = WHITE
        u.pre = NIL
    time = 0
    S = new empty stack
    for each u in G.V
        PUSH(S, u)
        while(!S.empty())
            TOP(S, s)
            if s.color == WHITE
                time = time + 1
                s.d = time
                s.color = GRAY
                v = G.Adj[s].next
                if v != NIL
                    if v.color == WHITE
                        time++
                        v.d = time
                        v.color = GRAY
                        v.pre = s
                        PUSH(S, v)
                else 
                    POP(S)
                    s.color = BLACK
                    s.f = time
